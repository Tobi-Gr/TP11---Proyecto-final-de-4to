CREATE PROCEDURE AgregarPeli
	@titulo VARCHAR(100),
	@sinopsis VARCHAR(500),
	@anio INT,
	@foto VARCHAR(500)
AS
BEGIN
	INSERT INTO Pelicula(titulo, sinopsis, anioLanzamiento, foto)
	VALUES (@titulo, @sinopsis, @anio, @foto)
END
